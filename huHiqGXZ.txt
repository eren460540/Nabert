-- Fungsi Cek Duit Otomatis di Leaderstats
local function getDuit()
    local stats = LocalPlayer:FindFirstChild("leaderstats")
    if stats then
        -- Mencari folder duit yang cocok (Coins / Cash / Money)
        local duitObj = stats:FindFirstChild("Coins") or stats:FindFirstChild("Cash") or stats:FindFirstChild("Money") or stats:FindFirstChild("Duit")
        if duitObj then return duitObj.Value end
    end
    return 0
end

-- LOOP AUTO FARM (Harvest & Sell)
task.spawn(function()
    while true do
        task.wait(0.3)
        if _G.AutoFarm and rootPart then
            for _, v in pairs(workspace:GetDescendants()) do
                if v:IsA("ProximityPrompt") and _G.AutoFarm then
                    local namaParent = string.lower(v.Parent.Name)
                    if string.find(namaParent, "fruit") or string.find(namaParent, "harvest") or v.ObjectText == "Harvest" or v.ActionText == "Harvest" then
                        rootPart.CFrame = v.Parent.CFrame * CFrame.new(0, 2, 0)
                        task.wait(0.2)
                        fireproximityprompt(v)
                        task.wait(0.1)
                    end
                end
            end
        end
    end
end)

-- TIMED AUTO SELL
task.spawn(function()
    while true do
        if _G.AutoFarm then
            task.wait(15)
            for _, obj in pairs(workspace:GetDescendants()) do
                if obj:IsA("Part") or obj:IsA("MeshPart") then
                    if string.find(string.lower(obj.Name), "sell") and rootPart and _G.AutoFarm then
                        local lastPos = rootPart.CFrame
                        rootPart.CFrame = obj.CFrame
                        task.wait(1.5)
                        rootPart.CFrame = lastPos
                        break
                    end
                end
            end
        else
            task.wait(1)
        end
    end
end)

-- LOOP AUTO BUY (Pet & Seeds)
task.spawn(function()
    while true do
        task.wait(2) -- Cek berkala tiap 2 detik biar ga spam berlebihan
        local duitSekarang = getDuit()

        -- A. Logika Auto Buy Legendary Pet
        if _G.AutoPet and duitSekarang >= _G.HargaPetLegendary then
            -- Memicu sistem beli (Bisa lewat ProximityPrompt Toko / Pet Egg)
            for _, v in pairs(workspace:GetDescendants()) do
                if v:IsA("ProximityPrompt") and (string.find(string.lower(v.Parent.Name), "legendary") or string.find(string.lower(v.Parent.Name), "egg")) then
                    fireproximityprompt(v)
                end
            end
            -- OPSI CADANGAN: Nembak Remote Event (Buka Delta Remote Spy buat mastiin nama Remotenya)
            local buyRemote = game:GetService("ReplicatedStorage"):FindFirstChild("BuyPet", true) or game:GetService("ReplicatedStorage"):FindFirstChild("PurchasePet", true)
            if buyRemote then buyRemote:FireServer("Legendary") end
        end

        -- B. Logika Auto Buy Mythic Seed
        if _G.AutoMythicSeed and duitSekarang >= _G.HargaSeedMythic then
            for _, v in pairs(workspace:GetDescendants()) do
                if v:IsA("ProximityPrompt") and string.find(string.lower(v.Parent.Name), "mythic") then
                    fireproximityprompt(v)
                end
            end
            local seedRemote = game:GetService("ReplicatedStorage"):FindFirstChild("BuySeed", true) or game:GetService("ReplicatedStorage"):FindFirstChild("PurchaseItem", true)
            if seedRemote then seedRemote:FireServer("MythicSeed") end
        end

        -- C. Logika Auto Buy Super Seed
        if _G.AutoSuperSeed and duitSekarang >= _G.HargaSeedSuper then
            for _, v in pairs(workspace:GetDescendants()) do
                if v:IsA("ProximityPrompt") and string.find(string.lower(v.Parent.Name), "super") then
                    fireproximityprompt(v)
                end
            end
            local seedRemote = game:GetService("ReplicatedStorage"):FindFirstChild("BuySeed", true) or game:GetService("ReplicatedStorage"):FindFirstChild("PurchaseItem", true)
            if seedRemote then seedRemote:FireServer("SuperSeed") end
        end
    end
end)

-- 3. KONTROL INTERAKSI TOMBOL (TOGGLE SYSTEM)
-- =========================================================

local function aturToggle(tombol, variabelGlobal, teksAsli)
    tombol.MouseButton1Click:Connect(function()
        _G[variabelGlobal] = not _G[variabelGlobal]
        if _G[variabelGlobal] then
            tombol.BackgroundColor3 = Color3.fromRGB(50, 150, 50) -- Hijau (ON)
            tombol.Text = teksAsli .. ": ON"
        else
            tombol.BackgroundColor3 = Color3.fromRGB(150, 50, 50) -- Merah (OFF)
            tombol.Text = teksAsli .. ": OFF"
        end
    end
end
